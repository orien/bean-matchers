1   package com.google.code.beanmatchers;
2   
3   import java.lang.reflect.Array;
4   import java.util.List;
5   import org.hamcrest.Description;
6   import org.hamcrest.TypeSafeDiagnosingMatcher;
7   
8   abstract class AbstractBeanToStringMatcher<T> extends TypeSafeDiagnosingMatcher<Class<T>> {
9     final TypeBasedValueGenerator valueGenerator;
10  
11    AbstractBeanToStringMatcher(TypeBasedValueGenerator valueGenerator) {
12      this.valueGenerator = valueGenerator;
13    }
14  
15    protected boolean toStringDescribesProperties(
16        Class beanType, List<String> properties, Description mismatchDescription) {
17      String toStringResult = new JavaBean(beanType).toString();
18      if (!toStringResult.contains(beanType.getSimpleName())) {
19        describeToStringMismatch(beanType, mismatchDescription)
20            .appendText("It did not produce the bean class name: ")
21            .appendValue(toStringResult);
22        return false;
23      }
24      for (String property : properties) {
25        JavaBean bean = new JavaBean(beanType);
26        Object value = valueGenerator.generate(bean.propertyType(property));
27        bean.setProperty(property, value);
28        toStringResult = bean.toString();
29        if (!toStringResult.contains(property)) {
30          describeToStringMismatch(beanType, mismatchDescription)
31              .appendText("It did not produce the property name ")
32              .appendValue(property)
33              .appendText(", actual output ")
34              .appendValue(toStringResult);
35          return false;
36        }
37        if (resultDoesNotContainValue(toStringResult, value)) {
38          describeToStringMismatch(beanType, mismatchDescription)
39              .appendText("It did not produce the value for property ")
40              .appendValue(property)
41              .appendText(", actual output ")
42              .appendValue(toStringResult);
43          return false;
44        }
45      }
46      return true;
47    }
48  
49    private boolean resultDoesNotContainValue(String toStringResult, Object value) {
50      if (value.getClass().isArray()) {
51        for (int i = 0; i < Array.getLength(value); i++) {
52          if (resultDoesNotContainValue(toStringResult, Array.get(value, i))) {
53            return true;
54          }
55        }
56        return false;
57      } else {
58        return !toStringResult.contains(value.toString());
59      }
60    }
61  
62    private Description describeToStringMismatch(Class beanType, Description mismatchDescription) {
63      return mismatchDescription
64          .appendText("bean of type ")
65          .appendValue(beanType.getName())
66          .appendText(" had an invalid toString() method. ");
67    }
68  }
