1   package com.google.code.beanmatchers;
2   
3   import org.hamcrest.Description;
4   import org.hamcrest.TypeSafeDiagnosingMatcher;
5   
6   import java.util.List;
7   
8   import static com.google.code.beanmatchers.ValueGenerators.generateTwoDistinctValues;
9   
10  abstract class AbstractBeanHashCodeMatcher<T> extends TypeSafeDiagnosingMatcher<Class<T>> {
11      final TypeBasedValueGenerator valueGenerator;
12  
13      AbstractBeanHashCodeMatcher(TypeBasedValueGenerator valueGenerator) {
14          this.valueGenerator = valueGenerator;
15      }
16  
17      protected boolean hashCodeIsInfluencedByProperties(JavaBean bean, List<String> properties, Description mismatchDescription) {
18          for (String property : properties) {
19              if (hashCodeNotInfluencedByProperty(bean, property)) {
20                  mismatchDescription
21                          .appendText("bean of type ")
22                          .appendValue(bean.beanType().getName())
23                          .appendText(" had a hashCode not influenced by the property ")
24                          .appendValue(property);
25                  return false;
26              }
27          }
28          return true;
29      }
30  
31      private boolean hashCodeNotInfluencedByProperty(JavaBean bean, String property) {
32          Class<?> propertyType = bean.propertyType(property);
33          DistinctValues values = generateTwoDistinctValues(valueGenerator, propertyType);
34          bean.setProperty(property, values.getValueOne());
35          int initialHashCode = bean.hashCode();
36          bean.setProperty(property, values.getValueTwo());
37          int influencedHashCode = bean.hashCode();
38          return initialHashCode == influencedHashCode;
39      }
40  }
