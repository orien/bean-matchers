3   final class ValueGenerators {
4     private static final int MAX_ATTEMPTS = 128;
5   
6     private ValueGenerators() {
7     }
8   
9     public static <T> DistinctValues<T> generateTwoDistinctValues(
10        TypeBasedValueGenerator valueGenerator, Class<T> valueType) {
11      int attempts = 0;
12      T valueOne = valueGenerator.generate(valueType);
13      T valueTwo;
14      do {
15        if (attempts == MAX_ATTEMPTS) {
16          throw new BeanMatchersException("Could not generate two distinct values after "
17              + MAX_ATTEMPTS + " attempts of type " + valueType.getName());
18        }
19        valueTwo = valueGenerator.generate(valueType);
20        attempts++;
21      } while (valueOne.equals(valueTwo));
22      return new DistinctValues<T>(valueOne, valueTwo);
23    }
