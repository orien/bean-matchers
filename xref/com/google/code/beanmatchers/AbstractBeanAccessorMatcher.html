1   package com.google.code.beanmatchers;
2   
3   import static com.google.code.beanmatchers.BeanOperations.getDeclaredMethod;
4   import static com.google.code.beanmatchers.BeanOperations.invokeMethod;
5   
6   import java.lang.reflect.Method;
7   import java.util.Arrays;
8   import java.util.List;
9   import java.util.Objects;
10  import org.hamcrest.Description;
11  import org.hamcrest.DiagnosingMatcher;
12  
13  abstract class AbstractBeanAccessorMatcher<T> extends DiagnosingMatcher<T> {
14  
15    final TypeBasedValueGenerator valueGenerator;
16  
17    AbstractBeanAccessorMatcher(TypeBasedValueGenerator valueGenerator) {
18      this.valueGenerator = valueGenerator;
19    }
20  
21    protected boolean beanHasValidGetterAndSetterForProperties(
22        JavaBean bean, List<String> properties, Description mismatchDescription) {
23      for (String property : properties) {
24        if (beanDoesNotHaveValidGetterAndSetterForProperty(bean, property)) {
25          mismatchDescription
26              .appendText("bean of type ")
27              .appendValue(bean.beanType().getName())
28              .appendText(" had an invalid getter/setter for the property ")
29              .appendValue(property);
30          return false;
31        }
32      }
33      return true;
34    }
35  
36    protected boolean beanDoesNotHaveValidGetterAndSetterForProperty(JavaBean bean, String property) {
37      try {
38        Class<?> propertyType = bean.propertyType(property);
39        Object testValue = valueGenerator.generate(propertyType);
40        bean.setProperty(property, testValue);
41        Object result = bean.getProperty(property);
42        boolean valid = Objects.equals(testValue, result);
43  
44        // for arrays fall back to compare element by element to allow for cloned arrays
45        if (!valid && propertyType.isArray()) {
46          valid = arraysEqual(propertyType, testValue, result);
47        }
48  
49        return !valid;
50      } catch (AccessorMissingException exception) {
51        return true;
52      }
53    }
54  
55    private boolean arraysEqual(Class<?> arrayType, Object testValue, Object result) {
56      Class<?> argumentType = arrayType.getComponentType().isPrimitive() ? arrayType : Object[].class;
57      Method method = getDeclaredMethod(Arrays.class, "equals", argumentType, argumentType);
58      return (boolean) invokeMethod(null, method, testValue, result);
59    }
60  }
