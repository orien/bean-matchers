1   package com.google.code.beanmatchers;
2   
3   import java.util.Arrays;
4   import java.util.List;
5   import java.util.Objects;
6   import org.hamcrest.Description;
7   import org.hamcrest.DiagnosingMatcher;
8   
9   abstract class AbstractBeanAccessorMatcher<T> extends DiagnosingMatcher<T> {
10  
11    final TypeBasedValueGenerator valueGenerator;
12  
13    AbstractBeanAccessorMatcher(TypeBasedValueGenerator valueGenerator) {
14      this.valueGenerator = valueGenerator;
15    }
16  
17    protected boolean beanHasValidGetterAndSetterForProperties(
18        JavaBean bean, List<String> properties, Description mismatchDescription) {
19      for (String property : properties) {
20        if (beanDoesNotHaveValidGetterAndSetterForProperty(bean, property)) {
21          mismatchDescription
22              .appendText("bean of type ")
23              .appendValue(bean.beanType().getName())
24              .appendText(" had an invalid getter/setter for the property ")
25              .appendValue(property);
26          return false;
27        }
28      }
29      return true;
30    }
31  
32    protected boolean beanDoesNotHaveValidGetterAndSetterForProperty(JavaBean bean, String property) {
33      try {
34        Class<?> propertyType = bean.propertyType(property);
35        Object testValue = valueGenerator.generate(propertyType);
36        bean.setProperty(property, testValue);
37        Object result = bean.getProperty(property);
38        boolean valid = Objects.equals(testValue, result);
39  
40        // for arrays fall back to compare element by element
41        // to allow cloned arrays in bean properties
42        if (!valid) {
43          if (propertyType.isArray()) {
44            if (propertyType.getComponentType().equals(Byte.TYPE)) {
45              valid = Arrays.equals((byte[]) testValue, (byte[]) result);
46            } else if (propertyType.getComponentType().equals(Short.TYPE)) {
47              valid = Arrays.equals((short[]) testValue, (short[]) result);
48            } else if (propertyType.getComponentType().equals(Integer.TYPE)) {
49              valid = Arrays.equals((int[]) testValue, (int[]) result);
50            } else if (propertyType.getComponentType().equals(Long.TYPE)) {
51              valid = Arrays.equals((long[]) testValue, (long[]) result);
52            } else if (propertyType.getComponentType().equals(Character.TYPE)) {
53              valid = Arrays.equals((char[]) testValue, (char[]) result);
54            } else if (propertyType.getComponentType().equals(Boolean.TYPE)) {
55              valid = Arrays.equals((boolean[]) testValue, (boolean[]) result);
56            } else if (propertyType.getComponentType().equals(Float.TYPE)) {
57              valid = Arrays.equals((float[]) testValue, (float[]) result);
58            } else if (propertyType.getComponentType().equals(Double.TYPE)) {
59              valid = Arrays.equals((double[]) testValue, (double[]) result);
60            } else {
61              valid = Arrays.equals((Object[]) testValue, (Object[]) result);
62            }
63          }
64        }
65        return !valid;
66      } catch (AccessorMissingException exception) {
67        return true;
68      }
69    }
70  }
