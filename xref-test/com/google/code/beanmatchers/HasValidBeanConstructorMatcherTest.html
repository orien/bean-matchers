18  public class HasValidBeanConstructorMatcherTest {
19  
20    private HasValidBeanConstructorMatcher unitUnderTest;
21  
22    @Mock
23    private Description descriptionMock;
24  
25    @BeforeMethod
26    public void setUp() {
27      unitUnderTest = new HasValidBeanConstructorMatcher();
28      initMocks(this);
29      when(descriptionMock.appendText(anyString())).thenReturn(descriptionMock);
30      when(descriptionMock.appendValue(any())).thenReturn(descriptionMock);
31    }
32  
33    @Test
34    public void beanWithNoArgsConstructorShouldMatch() {
35      // given
36      Class beanType = TestBeanWithOneProperty.class;
37  
38      // when
39      boolean match = unitUnderTest.matches(beanType);
40  
41      // then
42      assertThat(match, is(true));
43    }
44  
45    @Test
46    public void beanWithoutNoArgsConstructorShouldNotMatch() {
47      // given
48      Class beanType = TestBeanWithoutNoArgsConstructor.class;
49  
50      // when
51      boolean match = unitUnderTest.matches(beanType);
52  
53      // then
54      assertThat(match, is(false));
55    }
56  
57    @Test
58    public void beanWithInvalidGetterShouldBeDiagnosed() {
59      // given
60      Class beanType = TestBeanWithoutNoArgsConstructor.class;
61  
62      // when
63      unitUnderTest.matchesSafely(beanType, descriptionMock);
64  
65      // then
66      verify(descriptionMock).appendText("bean of type ");
67      verify(descriptionMock).appendValue(TestBeanWithoutNoArgsConstructor.class.getName());
68      verify(descriptionMock).appendText(" does not have a no-args constructor");
69    }
70  
71    @Test
72    public void shouldDescribeExpectation() {
73      // when
74      unitUnderTest.describeTo(descriptionMock);
75  
76      // then
77      verify(descriptionMock).appendText("bean class with a valid no-args constructor");
78    }
