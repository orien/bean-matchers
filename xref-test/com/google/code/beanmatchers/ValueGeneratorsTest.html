17  public class ValueGeneratorsTest {
18  
19    @Mock
20    private TypeBasedValueGenerator valueGeneratorMock;
21  
22    @BeforeMethod
23    public void setUp() {
24      initMocks(this);
25    }
26  
27    @Test
28    public void canGenerateTwoValues() {
29      // given
30      when(valueGeneratorMock.generate(Integer.class)).thenReturn(1, 2);
31  
32      // when
33      DistinctValues<Integer> result = generateTwoDistinctValues(valueGeneratorMock, Integer.class);
34  
35      // then
36      assertThat(result, is(notNullValue()));
37      assertThat(result.getValueOne(), is(equalTo(1)));
38      assertThat(result.getValueTwo(), is(equalTo(2)));
39    }
40  
41    @Test
42    public void shouldGenerateTwoDistinctValues() {
43      // given
44      when(valueGeneratorMock.generate(Integer.class)).thenReturn(1, 1, 2);
45  
46      // when
47      DistinctValues<Integer> result = generateTwoDistinctValues(valueGeneratorMock, Integer.class);
48  
49      // then
50      assertThat(result, is(notNullValue()));
51      assertThat(result.getValueOne(), is(equalTo(1)));
52      assertThat(result.getValueTwo(), is(equalTo(2)));
53    }
54  
55    @Test(expectedExceptions = BeanMatchersException.class)
56    public void shouldThrowExceptionGivenCannotGenerateTwoDistinctValuesAfterXAttempts() {
57      // given
58      when(valueGeneratorMock.generate(Integer.class)).thenReturn(1);
59  
60      // when
61      try {
62        generateTwoDistinctValues(valueGeneratorMock, Integer.class);
63      } catch (RuntimeException exception) {
64        verify(valueGeneratorMock, times(128 + 1)).generate(Integer.class);
65        throw exception;
66      }
67    }
68  }
